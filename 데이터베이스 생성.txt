CREATE OR REPLACE VIEW SMMEMBER AS
SELECT R.ID, C.CNAME, P.PNAME,
R.RCOUNT,P.PPRICE,R.RDATE
FROM 
SEMIRESERVE R,SEMIPRODUCT P, SEMICOMPANY C 
WHERE R.PNO=P.PNO AND R.CNO=C.CNO
ORDER BY R.ID ASC;

CREATE OR REPLACE VIEW SEMICOUNT AS
SELECT ID,COUNT(*) CNT
FROM SEMIRESERVE
GROUP BY ID
ORDER BY COUNT(*) DESC;
